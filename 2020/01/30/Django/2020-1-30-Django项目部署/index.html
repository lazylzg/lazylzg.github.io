Django 项目部署

安装 nginx

sudo apt-get install nginx

安装 mysql 数据库

配置虚拟环境

virtualenv venv

激活虚拟环境

source venv/bin/activate

安装 uwsgi

pip install uwsgi

sudo apt-get install mysql-server

生成 requirements.txt

pip freeze > requirements.txt

安装 requirements.txt

pip install -r requirements.txt

配置 nginx.conf

server {
    charset      utf-8;
    listen       443;
    server_name  api.domian.com;
    root /var/www/api;

    ssl on;
    ssl_certificate /var/www/api/1_api.domian.com_bundle.crt;
    ssl_certificate_key /var/www/api/2_api.domian.com.key;
    ssl_session_timeout 5m;
    ssl_protocols TLSv1 TLSv1.1 TLSv1.2;
    ssl_ciphers ECDHE-RSA-AES128-GCM-SHA256:HIGH:!aNULL:!MD5:!RC4:!DHE;
    ssl_prefer_server_ciphers on;

    location / {
        include   uwsgi_params;
        uwsgi_pass   127.0.0.1:8080;
    }
    # 指向django的media目录
    location /media  {
        alias /var/www/api/media; 
    }
    # 指向django的static目录
    location /static {
        alias /var/www/api/static; 
    }
}
server {
    listen 80;
    server_name course.zguolee.cn; 
    rewrite ^(.*)$ https://$host$1 permanent;
    location / {
        index index.html index.htm;
    }
}

配置 uwsgi.ini

[uwsgi]
# variables
projectname = projectname
projectdomain = api.domian.com
base = /var/www/api

# config
master = true
protocol = uwsgi
env = DJANGO_SETTINGS_MODULE=%(projectname).settings
pythonpath = %(base)/%(projectname)
module = %(projectname).wsgi
socket = 127.0.0.1:8080   
# 存进程id的文件
pidfile=uwsgi.pid
# 日志文件
daemonize=uwsgi.log

python manage.py collectstatic

service nginx start 启动 nginx 转发

uwsgi --ini uwsgi.ini
